# Python tooling required by the IDL codegen drivers and CI workflows.
# Install with: pip install -r requirements-codegen.txt
#
# Mintfile pins Swift tooling, Brewfile pins Homebrew formulas. This file
# declares the Python side so `idl/codegen/generate_*.sh` and
# idl-drift-check.yml can resolve the same versions on every contributor's
# machine. The *executable* source-of-truth for the protobuf cap and gRPC
# floor is `scripts/setup-toolchain.sh` — those constants
# (`PYTHON_PROTOBUF_EXPECTED`, `GRPCIO_TOOLS_EXPECTED`) drive the actual CI
# installs, and the ranges below MUST stay consistent with that script.
# Bump together with that script.

# Protobuf Python runtime — consumed by the IDL convenience drivers.
# Capped at <5 to match scripts/setup-toolchain.sh:163 (the IDL convenience
# generators are validated against protobuf 4.25.x; protobuf 5.x raised
# breaking-change advisories for our codegen path that have not yet been
# regression-tested).
protobuf>=4.25,<5

# gRPC Python codegen — generate_kotlin.sh / generate_ts.sh wrappers and
# the AsyncIterator client stubs introduced in GAP 09.
grpcio-tools>=1.65

# PyYAML — used by idl/codegen/swift-modality-abi/generate.py
pyyaml>=6.0
